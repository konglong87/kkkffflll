% Note: each section of this CV is stored in a two-column table. The
% left-hand column is narrow (1 inch) and stores dates. The right-hand
% column is wide (5.4 inches) and stores the main text.  Sections in
% which each entry might have multiple lines (e.g., Education) are put
% in a 'SectionTable' environment. Sections in which each entry might
% have just one line (e.g., Awards) are put in a 'SectionTableSingleSpace'
% environment. The only difference between the two environments is the line 
% spacing between each entry. Both environments take one argument, which 
% is the title of the section. See main document for how these
% environments are used.

\newcolumntype{R}[1]{>{\raggedleft\arraybackslash}p{#1}}                                                                                                                                                              
\newcolumntype{L}[1]{>{\raggedright\arraybackslash}p{#1}}

	\renewcommand*{\arraystretch}{1.6}
	\setlength{\tabcolsep}{10pt}
	\begin{longtable}[r]{R{1in} L{5.4in}} & #1 \\}
{\end{longtable}\vspace{-.3cm}}

\newenvironment{sectiontablesinglespace}[1]{

{\end{longtable}\vspace{-.3cm}}
